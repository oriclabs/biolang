Proteomics

BioLang provides tools for protein sequence analysis, mass spectrometry data processing, and structural feature extraction. These examples cover common proteomics workflows from basic sequence properties to peptide identification.

Protein Properties

Molecular weight and isoelectric point

# Calculate basic protein properties
let protein = protein"MKWVTFISLLFLFSSAYSRGVFRR"

let mw = protein |> molecular_weight
let pi = protein |> isoelectric_point
let charge = protein |> net_charge(ph: 7.4)
let gravy = protein |> gravy_score

print(f"Molecular weight: {mw |> round(2)} Da")
print(f"Isoelectric point: {pi |> round(2)}")
print(f"Net charge at pH 7.4: {charge |> round(2)}")
print(f"GRAVY score: {gravy |> round(3)}")

Amino acid composition

let protein = protein"MKWVTFISLLFLFSSAYSRGVFRRDAHKSEVAHRFK"

let composition = protein |> aa_composition
let aa_list = composition |> sort_by(|a, b| b.count - a.count)
for entry in aa_list {
  let pct = entry.count |> float / (protein |> len) |> float * 100.0
  print(f"{entry.aa}: {entry.count} ({pct |> round(1)}%)")
}

# Count hydrophobic residues (A, V, I, L, M, F, W, P)
let hydrophobic_aa = ["A", "V", "I", "L", "M", "F", "W", "P"]
let hydrophobic = to_string(protein) |> split("") |> filter(|aa| { hydrophobic_aa |> contains(aa) }) |> len
print(f"\nHydrophobic residues: {hydrophobic}")

Extinction coefficient

# Calculate extinction coefficient at 280nm
let seq = protein"MKWVTFISLLFLFSSAYSRGVFRRDAHKSEVAHRFKDLGE"

let ext_coeff = seq |> extinction_coefficient
let abs_01 = ext_coeff / (seq |> molecular_weight) * 10.0

print(f"Extinction coefficient: {ext_coeff} M^-1 cm^-1")
print(f"Abs 0.1% (1 g/L): {abs_01 |> round(3)}")

Protein Domain Search

Motif scanning

# Search for common protein motifs
let protein = fasta("protein.fa") |> first

# N-glycosylation sites: N-X-S/T (where X is not P)
let nglyc_sites = protein.seq |> find_motif("N[^P][ST]")
print(f"N-glycosylation sites: {nglyc_sites |> len}")
for site in nglyc_sites {
  print(f"  Position {site.start}: {site.match}")
}

# Phosphorylation sites (simplified)
let phos_sites = protein.seq |> find_motif("[ST]P")
print(f"Potential phosphorylation sites: {phos_sites |> len}")

# Signal peptide prediction (basic hydrophobicity check)
let signal = subseq(protein.seq, 0, 30)
let hydro = signal |> gravy_score
print(f"N-terminal hydrophobicity: {hydro |> round(3)} (>0.5 suggests signal peptide)")

Domain annotation from Pfam hits

# Parse Pfam domain hits
let domains = tsv("pfam_results.tsv")
  |> filter(|d| { d["e_value"] |> float < 1e-5 })
  |> sort_by(|a, b| a.start - b.start)

print("Significant domains found:")
for d in domains {
  print(f"  {d["domain_name"]} ({d["start"]}-{d["end"]}) E-value: {d["e_value"]}")
}

# Check for domain architecture
let has_kinase = domains |> any(|d| { d["domain_name"] |> contains("Kinase") })
let has_sh2 = domains |> any(|d| { d["domain_name"] |> contains("SH2") })
if has_kinase && has_sh2 {
  print("Receptor tyrosine kinase-like architecture detected")
}

Mass Spectrometry

In-silico tryptic digest

# Trypsin cleaves after K and R (except when followed by P)
let protein = protein"MKWVTFISLLFLFSSAYSRGVFRRDAHKSEVAHRFKDLGE"

let peptides = protein |> digest(enzyme: "trypsin", missed_cleavages: 1)

print("Tryptic peptides (up to 1 missed cleavage):")
for pep in peptides |> sort_by(|p| { p |> len }) {
  let mz = pep |> monoisotopic_mass / 1.0 + 1.00728  # [M+H]+
  print(f"  {pep} (m/z: {mz |> round(4)}, length: {pep |> len})")
}

Peptide mass fingerprint

# Generate theoretical peptide masses for database matching
let proteins = fasta("proteome.fa")

let mass_db = proteins
  |> flat_map(|p| {
    p.seq |> digest(enzyme: "trypsin", missed_cleavages: 0)
      |> filter(|pep| { pep |> len >= 6 && pep |> len <= 30 })
      |> map(|pep| { {
        protein: p.name,
        peptide: pep |> to_string,
        mass: pep |> monoisotopic_mass
      }})
  })
  |> sort_by(|a, b| a.mass - b.mass)

mass_db |> write_tsv("theoretical_masses.tsv")
print(f"Generated {mass_db |> len} theoretical peptide masses")

Matching observed masses to theoretical

# Match observed MS peaks to theoretical peptide masses
let observed = tsv("observed_peaks.tsv")
  |> map(|row| { row["mz"] |> float })

let theoretical = tsv("theoretical_masses.tsv")

let tolerance_ppm = 10.0

let matches = observed |> flat_map(|obs_mz| {
  theoretical
    |> filter(|t| {
      let ppm = ((obs_mz - t["mass"] |> float) / (t["mass"] |> float) * 1e6) |> abs
      ppm < tolerance_ppm
    })
    |> map(|t| { {
      observed_mz: obs_mz,
      theoretical_mass: t["mass"],
      protein: t["protein"],
      peptide: t["peptide"],
      ppm_error: ((obs_mz - t["mass"] |> float) / (t["mass"] |> float) * 1e6)
    }})
})

print(f"Matched {matches |> len} peptides to {matches |> map(|x| x.protein) |> unique |> len} proteins")
matches |> write_tsv("peptide_matches.tsv")

Protein Comparison

Pairwise sequence identity

# Compare protein sequences from a FASTA file
let proteins = fasta("family.fa") |> collect

for i in 0..proteins |> len {
  for j in (i + 1)..proteins |> len {
    let identity = align(proteins[i].seq, proteins[j].seq, matrix: "BLOSUM62")
      |> pct_identity
    print(f"{proteins[i].name} vs {proteins[j].name}: {identity |> round(1)}%")
  }
}

Multiple sequence alignment summary

# Analyze conservation from an MSA
let msa = fasta("aligned_proteins.fa")

let conservation = msa |> column_conservation
let n_conserved = conservation |> filter(|c| { c >= 0.9 }) |> len
let n_variable = conservation |> filter(|c| { c < 0.5 }) |> len

print(f"MSA columns: {conservation |> len}")
print(f"Highly conserved (>=90%): {n_conserved}")
print(f"Variable (<50%): {n_variable}")
print(f"Overall conservation: {conservation |> mean |> round(3)}")
