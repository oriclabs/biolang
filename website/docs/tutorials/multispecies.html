Multi-species Comparative Genomics

Compare genes and genomes across species to understand evolutionary relationships. This tutorial covers cross-species gene lookup, multiple sequence alignment, phylogenetic tree construction, and conservation analysis using BioLang's comparative genomics toolkit.

What you will learn

  • Looking up orthologous genes across species with ensembl_symbol()
  • Multiple sequence alignment with MUSCLE and MAFFT
  • Computing evolutionary distances (dN/dS ratios)
  • Building phylogenetic trees (neighbor-joining, maximum likelihood)
  • Conservation analysis across genomes
  • Visualizing conservation and divergence
Run this tutorial: Download multispecies.bl and run it with bl run examples/tutorials/multispecies.bl

Prerequisites

This tutorial assumes familiarity with the Querying Databases and Protein Analysis tutorials. You should be comfortable with sequences, tables, and database queries.

Step 1 — Looking Up Genes Across Species

Orthologs are genes in different species that evolved from a common ancestral gene through speciation. We can look up the same gene in multiple species using ensembl_symbol() to compare them. 

# comparative.bio — multi-species comparative genomics

# Define species and their gene symbols for TP53
# (gene symbols differ by species naming conventions)
let gene_lookups = [
  { species: "homo_sapiens",             symbol: "TP53" },
  { species: "mus_musculus",             symbol: "Trp53" },
  { species: "rattus_norvegicus",        symbol: "Tp53" },
  { species: "gallus_gallus",            symbol: "TP53" },
  { species: "danio_rerio",             symbol: "tp53" },
  { species: "drosophila_melanogaster",  symbol: "p53" },
]

# Look up TP53 orthologs in each species
let orthologs = gene_lookups |> map(|entry| {
  let gene = ensembl_symbol(entry.species, entry.symbol)
  { species: entry.species, gene_id: gene.id, symbol: gene.symbol,
    chromosome: gene.chromosome, start: gene.start, end: gene.end }
})

print("=== TP53 Orthologs ===")
for orth in orthologs {
  print(f"{orth.species}  {orth.gene_id}  {orth.symbol}")
}

# Fetch the protein sequences for each ortholog
let sequences = orthologs |> map(|o| {
  let seq = ensembl_sequence(o.gene_id, "protein")
  { name: o.species, gene: o.symbol, sequence: seq.seq }
})

for s in sequences {
  print(f"{s.name}: {len(s.sequence)} aa")
}

Step 2 — Multiple Sequence Alignment

# Align orthologs using MUSCLE
let alignment = align(sequences, {
  method:     "muscle",
  type:       "protein",
  iterations: 3,
})

print("=== Multiple Sequence Alignment ===")
print(f"Alignment length: {alignment.length}")
print(f"Sequences: {alignment.n_sequences}")
print(f"Gaps: {round(alignment.gap_fraction, 3)}")

# Print alignment with conservation markers
print_alignment(alignment, width: 80, conservation: true)
# homo_sapiens       MEEPQSDPSVEPPLSQETFSDLWKLLPENNV--LSPLPS...
# mus_musculus       MEESQAELSVEPPLSQETFSDLWKLLPENNV--LSPLPS...
# rattus_norvegicus  MEDSQAELVVEPPLSQETFSDLWKLLPENNV--LSPLPS...
# gallus_gallus      MEEPQSDLSAEPPLSQETFSDLWKLLPENNV--LSPLPS...
# danio_rerio        MDVQNDDLSVEPPLSQETFSDLWKLLPENNV--LGPLPS...
# drosophila_mel.    -----MDMSAEPALSRETFSDLWKLLPENNALGLSPLPS...
#                          * ::**:********* ******* .:******

# Conservation score per position
let conservation = conservation_scores(alignment)
print(f"Mean conservation: {round(mean(conservation), 3)}")

# Find highly conserved regions (>90% identity)
let conserved = conserved_regions(alignment, threshold: 0.9, min_length: 10)
print("\nHighly conserved regions:")
for region in conserved {
  print(f"  Position {region.start}-{region.end} ({region.length} residues, {round(region.identity * 100, 1)}%)")
}

Step 3 — Evolutionary Distances

# Compute pairwise distances from the alignment
let distances = distance_matrix(alignment, model: "jtt")

print("=== Pairwise Distances (JTT model) ===")
print(distances)

# For coding sequences, compute dN/dS (Ka/Ks) ratios
# Fetch the CDS sequences for each ortholog
let cds_sequences = orthologs |> map(|o| {
  let seq = ensembl_sequence(o.gene_id, "cds")
  { name: o.species, sequence: seq.seq }
})

# Codon-aware alignment
let codon_aln = align(cds_sequences, {
  method: "muscle",
  type:   "codon",   # align codons, not individual bases
})

# Compute dN/dS for all pairs
let dnds = pairwise_dnds(codon_aln, method: "yn00")

print("\n=== dN/dS Ratios ===")
for pair in dnds {
  let selection = match pair.omega {
    w if w > 1.0  => "positive",
    w if w < 0.5  => "purifying",
    _ => "neutral",
  }
  print(f"{pair.species1} vs {pair.species2}: dN/dS = {round(pair.omega, 4)} ({selection})")
}

# Site-specific dN/dS to find positively selected sites
let site_results = site_dnds(codon_aln, method: "slac")
let positive_sites = site_results |> filter(|s| s.omega > 1.0 and s.pvalue < 0.05)
print(f"\nPositively selected sites: {len(positive_sites)}")

Step 4 — Building a Phylogenetic Tree

# Neighbor-joining tree from distance matrix
let nj_tree = neighbor_joining(distances)

print("=== Phylogenetic Tree (Neighbor-Joining) ===")
print_ascii(nj_tree)
# ┌─── homo_sapiens
# ┌──┤
# │  └─── mus_musculus
# ──┤  ┌── rattus_norvegicus
# │  ┌──┤
# │  │  └── gallus_gallus
# └──┤
#    └──── danio_rerio
#       └── drosophila_melanogaster

# Write Newick format
write_newick(nj_tree, "results/tp53_tree.nwk")

# Maximum likelihood tree (more accurate but slower)
let ml_tree = maximum_likelihood(alignment, {
  model:      "JTT+G",     # JTT substitution model with gamma rates
  bootstrap:  100,          # 100 bootstrap replicates
})

print("\n=== Maximum Likelihood Tree ===")
print_ascii(ml_tree, show_bootstrap: true)

# Tree statistics
print(f"Tree length: {round(ml_tree.total_branch_length, 4)}")
print(f"Log-likelihood: {round(ml_tree.log_likelihood, 2)}")

# Save tree with bootstrap values
write_newick(ml_tree, "results/tp53_ml_tree.nwk")

# Generate publication-quality tree visualization
phylo_tree(ml_tree, {
  bootstrap:    true,
  branch_color: "conservation",  # color by evolutionary rate
  tip_labels:   gene_lookups |> map(|s| s.species),
  scale_bar:    true,
  title:        "TP53 Phylogeny",
  output:       "results/phylo_tree.svg",
})

Step 5 — Gene Family Expansion

# Look at how a gene family has expanded/contracted across species

# p53 family members: TP53, TP63, TP73
# Each species uses different symbol conventions
let family_queries = [
  { species: "homo_sapiens",    genes: ["TP53", "TP63", "TP73"] },
  { species: "mus_musculus",    genes: ["Trp53", "Trp63", "Trp73"] },
  { species: "rattus_norvegicus", genes: ["Tp53", "Tp63", "Tp73"] },
  { species: "gallus_gallus",   genes: ["TP53", "TP63", "TP73"] },
  { species: "danio_rerio",    genes: ["tp53", "tp63", "tp73"] },
  { species: "drosophila_melanogaster", genes: ["p53", "p63", "p73"] },
]

# Try to look up each gene; count successes
let family_data = family_queries |> map(|entry| {
  let counts = ["TP53", "TP63", "TP73"] |> enumerate() |> map(|i, label| {
    let symbol = entry.genes[i]
    # Attempt lookup; missing genes return nil
    let result = try { ensembl_symbol(entry.species, symbol) }
    { gene: label, found: result != nil }
  })
  { species: entry.species, genes: counts }
})

# Build a table
let family_table = family_data
  |> flat_map(|sp| sp.genes |> map(|g| {
    { species: sp.species, gene: g.gene, present: if g.found then 1 else 0 }
  }))
  |> to_table()
  |> pivot_wider(names_from: "gene", values_from: "present")

print("=== p53 Family Presence Across Species ===")
print(family_table)

# Visualize with a dot plot
plot(family_table |> pivot_longer(cols: ["TP53", "TP63", "TP73"], names_to: "gene", values_to: "present"), {
  x:      "gene",
  y:      "species",
  size:   "present",
  color:  "present",
  title:  "p53 Family Across Species",
  output: "results/family_size.svg",
})

Step 8 — Cross-Species Sequence Comparison

# Compare protein sequences between human and each other species

fn pairwise_comparison(species1_entry, species2_entry) -> Map {
  let seq1 = ensembl_sequence(species1_entry.gene_id, "protein")
  let seq2 = ensembl_sequence(species2_entry.gene_id, "protein")

  let pair_aln = align([
    { name: species1_entry.species, sequence: seq1.seq },
    { name: species2_entry.species, sequence: seq2.seq },
  ], { method: "muscle", type: "protein" })

  let cons = conservation_scores(pair_aln)

  {
    species1:       species1_entry.species,
    species2:       species2_entry.species,
    length1:        len(seq1.seq),
    length2:        len(seq2.seq),
    aln_length:     pair_aln.length,
    mean_identity:  round(mean(cons), 3),
    gap_fraction:   round(pair_aln.gap_fraction, 3),
  }
}

# Compare human (first entry) with each other species
let human = orthologs[0]
let comparisons = orthologs[1..] |> map(|sp|
  pairwise_comparison(human, sp)
) |> to_table()

print("=== Pairwise Protein Comparison vs Human ===")
print(comparisons)

Step 9 — Multi-Gene Phylogenomics

# Build a species tree from multiple genes (concatenation approach)

# Gene symbols per species for our marker genes
let marker_genes = [
  { label: "TP53",  symbols: { homo_sapiens: "TP53",  mus_musculus: "Trp53", rattus_norvegicus: "Tp53",
                                gallus_gallus: "TP53",  danio_rerio: "tp53" } },
  { label: "BRCA1", symbols: { homo_sapiens: "BRCA1", mus_musculus: "Brca1", rattus_norvegicus: "Brca1",
                                gallus_gallus: "BRCA1", danio_rerio: "brca1" } },
  { label: "RB1",   symbols: { homo_sapiens: "RB1",   mus_musculus: "Rb1",   rattus_norvegicus: "Rb1",
                                gallus_gallus: "RB1",   danio_rerio: "rb1" } },
  { label: "APC",   symbols: { homo_sapiens: "APC",   mus_musculus: "Apc",   rattus_norvegicus: "Apc",
                                gallus_gallus: "APC",   danio_rerio: "apc" } },
  { label: "PTEN",  symbols: { homo_sapiens: "PTEN",  mus_musculus: "Pten",  rattus_norvegicus: "Pten",
                                gallus_gallus: "PTEN",  danio_rerio: "ptena" } },
]

let target_species = ["homo_sapiens", "mus_musculus", "rattus_norvegicus",
                      "gallus_gallus", "danio_rerio"]

# Fetch and align each gene across species
let alignments = marker_genes |> map(|gene| {
  print(f"Aligning {gene.label}...")
  let seqs = target_species |> map(|sp| {
    let symbol = gene.symbols[sp]
    let info = ensembl_symbol(sp, symbol)
    let seq = ensembl_sequence(info.id, "protein")
    { name: sp, sequence: seq.seq }
  })
  align(seqs, method: "mafft")
})

# Concatenate alignments into a supermatrix
let supermatrix = concatenate_alignments(alignments)
print(f"Supermatrix: {supermatrix.n_sequences} taxa, {supermatrix.length} sites")

# Partition information for model selection
let partitions = alignments |> enumerate() |> map(|i, aln| {
  { name: marker_genes[i].label, start: aln.start_in_concat, end: aln.end_in_concat, model: "JTT+G" }
})

# Build partitioned ML tree
let species_tree = maximum_likelihood(supermatrix, {
  partitions: partitions,
  bootstrap:  200,
})

print("\n=== Species Tree ===")
print_ascii(species_tree, show_bootstrap: true)

phylo_tree(species_tree, {
  bootstrap:  true,
  tip_labels: target_species,
  title:      "Species Tree (5-gene phylogenomics)",
  output:     "results/species_tree.svg",
})

Step 10 — Complete Comparative Pipeline

# comparative_pipeline.bio — full comparative genomics workflow

fn comparative_analysis(
  gene_symbols: Map,  # { species_name: symbol }
  gene_label: String,
  output_dir: String,
) {
  print(f"=== Comparative Analysis: {gene_label} ===\n")

  # 1. Look up gene in each species
  let species_list = gene_symbols |> keys()
  let gene_info = species_list |> map(|sp| {
    let info = ensembl_symbol(sp, gene_symbols[sp])
    { species: sp, gene_id: info.id, symbol: info.symbol }
  })
  print(f"Found gene in {len(gene_info)} species")

  # 2. Fetch protein sequences and align
  let protein_seqs = gene_info |> map(|g| {
    let seq = ensembl_sequence(g.gene_id, "protein")
    { name: g.species, sequence: seq.seq }
  })
  let aln = align(protein_seqs, method: "muscle")
  write_fasta(aln, f"{output_dir}/{gene_label}_alignment.fa")

  # 3. Build tree
  let tree = maximum_likelihood(aln, { model: "JTT+G", bootstrap: 100 })
  write_newick(tree, f"{output_dir}/{gene_label}_tree.nwk")
  phylo_tree(tree, { output: f"{output_dir}/{gene_label}_tree.svg", bootstrap: true })

  # 4. Conservation analysis
  let cons = conservation_scores(aln)
  plot(cons |> enumerate() |> to_table(), {
    output: f"{output_dir}/{gene_label}_conservation.svg",
    title: f"{gene_label} Conservation",
  })

  # 5. dN/dS analysis
  let cds_seqs = gene_info |> map(|g| {
    let seq = ensembl_sequence(g.gene_id, "cds")
    { name: g.species, sequence: seq.seq }
  })
  let codon_aln = align(cds_seqs, method: "muscle", type: "codon")
  let dnds = pairwise_dnds(codon_aln)
  dnds |> to_table() |> write_csv(f"{output_dir}/{gene_label}_dnds.csv")

  # 6. Summary
  let summary = {
    gene:           gene_label,
    n_species:      len(gene_info),
    aln_length:     aln.length,
    mean_cons:      round(mean(cons), 3),
    tree_length:    round(tree.total_branch_length, 4),
  }

  write_json(f"{output_dir}/{gene_label}_summary.json", summary)
  print(f"Results saved to {output_dir}/")
}

# Run for TP53
let tp53_symbols = {
  homo_sapiens: "TP53", mus_musculus: "Trp53", rattus_norvegicus: "Tp53",
  gallus_gallus: "TP53", danio_rerio: "tp53", drosophila_melanogaster: "p53",
}
comparative_analysis(tp53_symbols, "TP53", "results/comparative")
